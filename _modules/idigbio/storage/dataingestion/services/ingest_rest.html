[docs]    def GET(self):
        """
        Get ingestion status.
        """
        try:
            total, remaining = ingest_service.check_progress()
            return json.dumps(dict(total=total, remaining=remaining))
        except ClientManagerException as ex:
            raise cherrypy.HTTPError(409, str(ex))
