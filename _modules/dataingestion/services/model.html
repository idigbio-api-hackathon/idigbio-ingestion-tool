[docs]def setup(db_file):
    global session
    
    db_conn = "sqlite:///%s" % db_file
    logger.info("DB Connection: %s" % db_conn)
    engine = create_engine(db_conn, connect_args={'check_same_thread':False})
    engine.Echo = True
    Base.metadata.create_all(engine)
    
    Session = sessionmaker(bind=engine)
    session = Session()



[docs]def md5_file(f, block_size=2**20):
    md5 = hashlib.md5()
    while True:
        data = f.read(block_size)
        if not data:
            break
        md5.update(data)
    return md5.hexdigest()



[docs]def add_image(batch, path):
    if session is None:
        raise ValueError
    with open(path, 'rb') as f:
        md5 = md5_file(f)
    
    record = ImageRecord(path, md5, batch)
    session.add(record)
    logger.debug("Image %s added to DB session." % path)
    return record
